\section{Subtype}
\begin{enumerate}
	\item[1)]
	Vi opskriver koden for en 4-bit subtractor som ses i kode \ref{lst:SubtractorCode}
	\begin{lstlisting}[caption={Subtractor kode},label={lst:SubtractorCode}]
	library ieee;
	use ieee.std_logic_1164.all;
	
	entity Subtypes is
	port (a,b : in std_logic;
	c  : out std_logic);
	end Subtypes;
	
	architecture dataflow of Subtypes is
	subtype bool is std_logic range '1' to 'Z';
	signal tmp : bool;
	begin 
	tmp<= 'U';
	c<= b and tmp;
	end dataflow;
		\end{lstlisting}
	\item[2)]

library ieee;
use ieee.std_logic_1164.all;

entity Subtypes is
port (a,b : in std_logic;
c  : out std_logic);
end Subtypes;

architecture dataflow of Subtypes is
subtype bool is std_logic range 'U' to 'Z';
signal tmp : bool;
begin 
tmp<= 'U';
c<= b and tmp;
end dataflow;
	\end{lstlisting}
